Arduino


							int count1 = 0;
							int count2 = 0;
							void setup() {
								pinMode(LED1, OUTPUT);
								pinMode(LED2, OUTPUT);
							}
							void loop() {
								digitalWrite(LED1,
									count1 < 1000);
								if(++count1 >= 2000)
									count1 = 0;
								digitalWrite(LED2,
									count2 < 1234);
								if(++count2 >= 2468)
									count2 = 0;
								delay(1);
							}
